# NopeNet Application Requirements
# ---------------------------------
# This file contains all the dependencies needed to run the NopeNet application
# For Python backend requirements, see server/requirements.txt

# Node.js requirements
# -------------------
# These packages should be installed using npm:
# 
# npm install --save react-markdown remark-gfm @tailwindcss/typography
#
# Main dependencies:
# - next.js: 14.0.4
# - react: 18.x
# - tailwindcss: 3.3.0+
# - framer-motion: 10.18.0+
# - ai: 2.2.31+
# - react-markdown: latest
# - remark-gfm: latest
# - @tailwindcss/typography: latest
#
# Development dependencies:
# - typescript: 5.x
# - eslint: 8.x
# - concurrently: 8.2.0+
#
# See package.json for the complete list of dependencies

# Python requirements
# ------------------
# The backend server requires Python 3.8+ and the packages listed in server/requirements.txt
# Install using: pip install -r server/requirements.txt

# System requirements
# ------------------
# - Node.js v16 or higher
# - Python v3.8 or higher
# - npm or yarn

# Installation
# -----------
# 1. Install frontend dependencies:
#    npm install
#
# 2. Install backend dependencies:
#    pip install -r server/requirements.txt
#
# 3. Build the frontend:
#    npm run build
#
# 4. Start the application:
#    npm run dev
